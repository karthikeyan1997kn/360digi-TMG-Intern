import pandas as pd

# Assuming you have data in a list of dictionaries, where each dictionary represents a row of data
data = [
    {'Typeofsales': 'Sale', 'Final_Sales': 100.0},
    {'Typeofsales': 'Sale', 'Final_Sales': 150.0},
    {'Typeofsales': 'Sale', 'Final_Sales': 200.0},
    {'Typeofsales': 'Sale', 'Final_Sales': 150.0},
    {'Typeofsales': 'Sale', 'Final_Sales': 250.0}
]

# Create a Pandas DataFrame
df = pd.DataFrame(data)

# Calculate the mean, median, and mode

median_final_sales = df['Final_Sales'].median()
mode_final_sales = df['Final_Sales'].mode().values[0]
